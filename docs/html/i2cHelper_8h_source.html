137/// @brief Generate I2C START condition
138/// @param dev Pointer to I2C device
139static inline __attribute__((always_inline))
140void i2cStartCondition(i2cDev_t * dev){
141
142 __i2cDelay(250000/(__i2cFreq(dev)));
143
144 /// Release both SCL, SDA line
145 __i2cSetSCL(dev, HIGH);
146 __i2cSetSDA(dev, HIGH);
147
148 /// Wait for both SCL and SDA are stable (a half of SCL period)
149 __i2cDelay(500000/(__i2cFreq(dev)));
150
151 /// Make start condition (SDA down to LOW when SCL is HIGH)
152 __i2cSetSDA(dev, LOW);
153
154 /// Wait for a quarter of SCL period
155 __i2cDelay(250000/(__i2cFreq(dev)));
156
157 /// Pull SCL down for 1st clock pulse
158 __i2cSetSCL(dev, LOW);
159}

185/// @brief Send I2C address frame with R/W bit
186/// @param dev Pointer to I2C device
187/// @param addr7 7-bit slave address
188/// @param rw Read/Write mode (I2C_WRITE_MODE or I2C_READ_MODE)
189static inline __attribute__((always_inline))
190void i2cSendAddressFrame(i2cDev_t * dev, uint8_t addr7, uint8_t rw){
191
192 uint8_t addressFrame = 0;
193
194 if((rw & 1U) == I2C_WRITE_MODE){
195 /// WRITE MODE
196 addressFrame = (addr7 << 1);
197 }else{
198 /// SLAVE MODE
199 addressFrame = (addr7 << 1) | 1U;
200 }
201 // __i2cLog1("addressFrame = 0x%04x", addressFrame);
202 /// Start sending 8 bit of address frame from MSB to LSB
203 for(uint8_t mask = 0x80; mask; mask >>= 1){
204
205 /// Export bit data to SDA
206 __i2cSetSDA(dev, boolCast(mask & addressFrame));
207 /// Make mono pulse of SCL
208 __i2cDelay(500000/__i2cFreq(dev));
209 __i2cSetSCL(dev, HIGH);
210 __i2cDelay(500000/__i2cFreq(dev));
211 __i2cSetSCL(dev, LOW);
212 }
213}
