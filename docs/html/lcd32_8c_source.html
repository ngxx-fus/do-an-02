80def __lcd32ConfigCanvas(lcd32Canvas_t *canvas, dim_t maxRow, dim_t maxCol) {
81 lcd32Entry("__lcd32ConfigCanvas(%p, %d, %d)", canvas, maxRow, maxCol);
82 lcd32NULLCheck(canvas, STR(canvas), STR(__lcd32ConfigCanvas), return ERR_NULL;);
83
84 canvas->maxRow = maxRow;
85 canvas->maxCol = maxCol;
86
87 // Free old buffer if it exists
88 if (__isnot_null(canvas->buff)) {
89 lcd32Warn("[__lcd32ConfigCanvas] canvas->buff = %p", canvas->buff);
90 __lcd32DeleteCanvasBuffer(canvas);
91 }
92
93 // Allocate array of row pointers
94 canvas->buff = (color_t **) heap_caps_malloc(sizeof(color_t *) * maxRow, MALLOC_CAP_SPIRAM);
95 if (!canvas->buff) {
96 lcd32Err("lcd32", "PSRAM allocation failed for row pointers array!");
97 return ERR_PSRAM_FAILED;
98 }
99
100 // Allocate each row buffer
101 for (dim_t r = 0; r < maxRow; ++r) {
102 canvas->buff[r] = (color_t *) heap_caps_malloc(sizeof(color_t) * maxCol, MALLOC_CAP_SPIRAM);
103 if (!canvas->buff[r]) {
104 // Free previously allocated rows
105 for (dim_t i = 0; i < r; ++i) free(canvas->buff[i]);
106 free(canvas->buff);
107 canvas->buff = NULL;
108 lcd32Err("lcd32", "PSRAM allocation failed for row %d!", r);
109 return ERR_PSRAM_FAILED;
110 }
111 memset(canvas->buff[r], 0, sizeof(color_t) * maxCol);
112 }
113
114 lcd32Log("Canvas created successfully in PSRAM (%d x %d).", maxRow, maxCol);
115 lcd32Exit("__lcd32ConfigCanvas -> OKE");
116 return OKE;
117}

284 def devId = 0;
285 devId = __lcd32ContReadRegisterStart(dev, ILI9341_READ_ID4) & 0xFF;
286 lcd32Log("Device ID (1st read): 0x%4x", devId);

436def __lcd32SetAddressWindow(lcd32Dev_t *dev, dim_t row, dim_t col, dim_t height, dim_t width) {
437 lcd32Entry("__lcd32SetAddressWindow(%p, r=%u, c=%u, h=%u, w=%u)", dev, row, col, height, width);
438
439 dim_t row1 = row;
440 dim_t row2 = row + height - 1;
441 dim_t col1 = col;
442 dim_t col2 = col + width - 1;
443
444 //--- Column Address Set (0x2A) ---
445 __lcd32WriteCommand(dev, ILI9341_COLUMN_ADDRESS_SET);
446 __lcd32WriteData(dev, col1 >> 8);
447 __lcd32WriteData(dev, col1 & 0xFF);
448 __lcd32WriteData(dev, col2 >> 8);
449 __lcd32WriteData(dev, col2 & 0xFF);
450
451 //--- Page Address Set (0x2B) ---

453 __lcd32WriteData(dev, row1 >> 8);
454 __lcd32WriteData(dev, row1 & 0xFF);

474def lcd32FlushCanvas(lcd32Dev_t *dev) {
475 lcd32Entry("lcd32FlushCanvas(%p)", dev);
476
477 // Check for null pointer
478 lcd32NULLCheck(dev, STR(dev), STR(lcd32FlushCanvas), return ERR_NULL;);
479 lcd32Canvas_t *canvas = &dev->canvas;
480
481 //--- Configure write region: full-screen area ---
482 // Reuse the address window API to set the region (0,0) (maxRow,maxCol)
483 __lcd32SetAddressWindow(dev, 0, 0, canvas->maxRow, canvas->maxCol);
484
485 //--- Send command to start memory write ---
486 __lcd32WriteCommand(dev, ILI9341_MEMORY_WRITE);
487
488 //--- Flush the entire canvas buffer to the LCD ---
489 __lcd32SetDataTransaction(dev);
490 __lcd32StartTransaction(dev);
491 __lcd32SetReadPin(dev, 1);
492 for (dim_t r = 0; r < canvas->maxRow; ++r) {
493 for (dim_t c = 0; c < canvas->maxCol; ++c) {
494 __lcd32SetParallelData(dev, canvas->buff[r][c]);
495 __lcd32SetWritePin(dev, 0);
496 __lcd32SetWritePin(dev, 1);
497 }
498 }
499 __lcd32StopTransaction(dev);
500 lcd32Exit("lcd32FlushCanvas() : OKE");
501 return OKE;
502}

519def lcd32DrawEmptyRect(lcd32Dev_t *lcd, dim_t rTopLeft, dim_t cTopLeft, dim_t rBottomRight, dim_t cBottomRight, dim_t edgeSize, color_t color){
520 if (edgeSize < 1) return OKE;
521
522 if (rTopLeft < 0 || rTopLeft >= lcd->canvas.maxRow ||
523 cTopLeft < 0 || cTopLeft >= lcd->canvas.maxCol ||
524 rBottomRight < 0 || rBottomRight >= lcd->canvas.maxRow ||
525 cBottomRight < 0 || cBottomRight >= lcd->canvas.maxCol)
526 return ERR_INVALID_ARG;
527
528 if (rTopLeft > rBottomRight || cTopLeft > cBottomRight)
529 return ERR_INVALID_ARG;
530
531 if (rBottomRight - rTopLeft <= 1 || cBottomRight - cTopLeft <= 1)
532 return OKE;
533
534 // Draw vertical edges
535 for (dim_t x = rTopLeft; x <= rBottomRight; ++x) {
536 __lcd32SetPixel(lcd, x, cTopLeft, color);
537 __lcd32SetPixel(lcd, x, cBottomRight, color);
538 }
539
540 // Draw horizontal edges
541 for (dim_t y = cTopLeft; y <= cBottomRight; ++y) {
542 __lcd32SetPixel(lcd, rTopLeft, y, color);
543 __lcd32SetPixel(lcd, rBottomRight, y, color);
544 }
545
546 // Recursive shrink for thicker borders
547 return lcd32DrawEmptyRect(lcd,
548 rTopLeft + 1, cTopLeft + 1,
549 rBottomRight - 1, cBottomRight - 1,
550 edgeSize - 1, color);
