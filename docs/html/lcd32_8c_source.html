17def shiftArrayLeft(uint8_t * arr, size_t size, uint8_t newLSB) {
18 if(__is_null(arr)){

24 return ERR;

26 uint8_t carry = 0;
27 for (size_t i = 0; i < size; i++) {
28 uint8_t next_carry = (arr[i] & 0x80) ? 1 : 0;
29 arr[i] = (uint8_t)((arr[i] << 1) | carry);
30 carry = next_carry;

33 return OKE;

53 uint8_t next_carry = (arr[i] & 0x01) ? 1 : 0;
