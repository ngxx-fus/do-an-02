244def oledSetPixel(oled128x64Dev_t *oled, xy_t row, xy_t col, color_t c) {
245 __oledLog1("** oledSetPixel(%p, %d, %d)", oled, row, col);
246
247 // Validate coordinates
248 if (row < 0 || row >= oled->height || col < 0 || col >= oled->width)
249 return ERR_INVALID_ARG;
250
251 // Each page = 8 rows (07)
252 xy_t pageIndex = row >> 3;
253 xy_t bitIndex = row & 0x7;
254
255 // Compute byte index in framebuffer
256 xy_t canvasIndex = pageIndex * oled->width + col;
257
258 // Write bit in framebuffer
259 if (c)
260 oled->canvas[canvasIndex] |= __mask8(bitIndex);
261 else
262 oled->canvas[canvasIndex] &= __inv_mask8(bitIndex);
263
264 return OKE;
265}

270 xy_t edgeSize, color_t color){

293 oledSetPixel(oled, rTopLeft, y, color);
294 oledSetPixel(oled, rBottomRight, y, color);
295 }
296
297 // Recursive shrink
298 return oledDrawEmptyRect(oled,
299 rTopLeft + 1, cTopLeft + 1,
300 rBottomRight - 1, cBottomRight - 1,

304def oledDrawChar(oled128x64Dev_t *oled, xy_t r, xy_t y, char c, color_t cl) {
305 // __oledEntry("oled_draw_char(%d, %d, %c, %d)", r, y, c, cl);
306 if (c < 0x20 || c > 0x7E) {
307 __oledExit("oledDrawChar() : ESP_ERR_INVALID_ARG");
308 return ESP_ERR_INVALID_ARG;
309 }
310
311 GFXglyph glyph = mainFont.glyph[c - 0x20];
312
313 uint16_t bo = glyph.bitmapOffset;
314 xy_t w = glyph.width;
315 xy_t h = glyph.height;
316 xy_t xo = glyph.xOffset;
317 xy_t yo = glyph.yOffset;
318
319 xy_t bitmap_top_x = r + yo;
320 xy_t bitmap_left_y = y + xo;
321 byte_t bits = 0;
322 byte_t bitCount = 0;
323
324 REPT(xy_t, yy, 0, h) {
325 REPT(xy_t, xx, 0, w) {
326 if (bitCount == 0) {
327 bits = mainFont.bitmap[bo++];
328 bitCount = 8;
329 }
330 if (bits & 0x80) {
331 xy_t pixel_x = bitmap_top_x + yy;
332 xy_t pixel_y = bitmap_left_y + xx;
333
334 if (pixel_x >= __min_x && pixel_x < (oled->height) &&
335 pixel_y >= __min_y && pixel_y < (oled->width)) {
336 oledSetPixel(oled, pixel_x, pixel_y, cl);
337 }
338 }
339 bits <<= 1;
340 bitCount--;
341 }
342 }
343
344 // __oledExit("oledDrawChar() : OKE");
345 return OKE;
346}

348def oledDrawText(oled128x64Dev_t *oled, xy_t r, xy_t c,
349 const char* str, color_t color, bool wrap, byte_t lineHeight)
350{
351 __oledEntry("oledDrawText(%d, %d, \"%s\", %d, wrap=%d, lh=%d)",
352 r, c, str, color, wrap, lineHeight);
353
354 if (!oled || !str)
355 return ERR_INVALID_ARG;
356
357 xy_t cursor_r = r;
358 xy_t cursor_c = c;
359 byte_t finalLineHeight = __max(lineHeight, defaultTextH(mainFont));
360
361 while (*str) {
362 char ch = *str++;
363
364 if (ch == '\n') {
365 cursor_c = 0;
366 cursor_r += finalLineHeight;
367 continue;
368 }
369
370 if (ch < 0x20 || ch > 0x7E)
371 continue;
372
373 GFXglyph glyph = mainFont.glyph[ch - 0x20];
374
375 // Wrap if needed
376 if (wrap && (cursor_c + glyph.xAdvance > oled->width)) {
377 cursor_c = 0;
378 cursor_r += finalLineHeight;
379 }
380
381 oledDrawChar(oled, cursor_r, cursor_c, ch, color);
382 cursor_c += glyph.xAdvance;
383
384 }
385
386 __oledExit("oledDrawText() : OKE");
387 return OKE;
388}
