221 __I2C_ASSERT_DEV_OK1(STR(i2cSendByte), dev, STR(dev));
222
223 if(__hasFlagBitClr(extConf, I2C_DIS_ENTRY_CRITICAL_SEC)) vPortEnterCritical(&(dev->mutex));
224
226 if(__hasFlagBitClr(extConf, I2C_DIS_START_CONDI)) i2cStartCondition(dev);
228 i2cSendAddressFrame(dev, addr7, rw);
230 if(i2cGetReturnBit(dev) == I2C_NACK) {
231 __i2cErr("[i2cSendByte] Got NACK after send address frame!");
232 goto i2cSendByte_ReturnERR;
233 }
235 i2cSendDataFrame(dev, byte);
237 if(i2cGetReturnBit(dev) == I2C_NACK) {
238 __i2cErr("[i2cSendByte] Got NACK after send byte high!");
239 goto i2cSendByte_ReturnERR;
240 }
