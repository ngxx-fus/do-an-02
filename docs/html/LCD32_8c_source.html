123DefaultRet_t LCD32Init(LCD32Dev_t * Dev){
124 LCD32Entry("LCD32Init(%p)", Dev);
125 DefaultRet_t ret;
126 P16Dev_t * P16Dev = &(Dev->P16Com);
127
129 ret = P16ComInit(P16Dev);
130 if(ret != STAT_OKE){
131 LCD32ReturnWithLog(ret, "LCD32Init() : Error");
132 }
133
135 if(IsValidPin(Dev->BrightLight)){
136 IOConfigAsOutput(Mask64(Dev->BrightLight), -1, -1);
137 LCD32SetHighBrightLightPin(Dev); // Turn off/Default state
138 }
139
141 P16SetLowResetPin(P16Dev);
142 P16BlockingDelay(10000); // 10ms
143 P16SetHighResetPin(P16Dev);
144 P16BlockingDelay(120000); // 120ms wait after reset
145
147 P16SetHighChipSelPin(P16Dev);
148 P16SetHighReadPin(P16Dev);
149 P16SetHighWritePin(P16Dev);
150
151 // VERIFY DEVICE ID (0xD3 Read ID4)
152 LCD32StartTransaction(Dev);
153
154 // Command 0xD3: Read ID4
155 LCD32WriteCmd(Dev, 0xD3);
156
157 // Switch to Data Mode (D/CX = 1) for reading parameters
158 LCD32SetDataTransaction(Dev);
159
160 // Read Sequence: Dummy -> Dummy (00) -> ID High (93) -> ID Low (41)
161 // Note: P16ComRead handles direction switching and strobe logic
162 volatile P16Data_t d1 = P16ComRead(P16Dev); // Dummy
163 volatile P16Data_t d2 = P16ComRead(P16Dev); // Dummy (Should be 0x00)
164 volatile P16Data_t d3 = P16ComRead(P16Dev); // ID High (Should be 0x93)
165 volatile P16Data_t d4 = P16ComRead(P16Dev); // ID Low (Should be 0x41)
166
167 LCD32StopTransaction(Dev);
168
169 uint16_t devId = ((d3 & 0xFF) << 8) | (d4 & 0xFF);
170 LCD32Log("[LCD32Init] Read ID: 0x%04X (Raw: %02X %02X %02X %02X)", devId, d1, d2, d3, d4);
171
172 if (devId != 0x9341) {
173 LCD32Err("[LCD32Init] Invalid Device ID: 0x%04X (Expected 0x9341)", devId);
174 // LCD32ReturnWithLog(STAT_ERR_UNSUPPORTED, "LCD32Init() : Wrong Device ID");
175 // Uncomment above line to enforce check, currently just logging error for debugging
176 }
177
178 // Set MADCTL based on orientation and update Width/Height
179 uint16_t madctl_val = 0;
180 switch (Dev->Orientation) {
181 case LCD32_ORIENTATION_PORTRAIT:
182 madctl_val = (MADCTL_MX | MADCTL_BGR); // 0x48
183 Dev->Width = LCD32_NATIVE_W;
184 Dev->Height = LCD32_NATIVE_H;
185 break;
186
187 case LCD32_ORIENTATION_LANDSCAPE:
188 madctl_val = (MADCTL_MV | MADCTL_BGR); // 0x28
189 Dev->Width = LCD32_NATIVE_H;
190 Dev->Height = LCD32_NATIVE_W;
191 break;
192
193 case LCD32_ORIENTATION_PORTRAIT_FLIP:
194 madctl_val = (MADCTL_MY | MADCTL_BGR); // 0x88
195 Dev->Width = LCD32_NATIVE_W;
196 Dev->Height = LCD32_NATIVE_H;
197 break;
198
199 case LCD32_ORIENTATION_LANDSCAPE_FLIP:
200 madctl_val = (MADCTL_MY | MADCTL_MX | MADCTL_MV | MADCTL_BGR); // 0xE8
201 Dev->Width = LCD32_NATIVE_H;
202 Dev->Height = LCD32_NATIVE_W;
203 break;
204
205 default: // Default to landscape as it's most common for these displays
206 madctl_val = (MADCTL_MV | MADCTL_BGR); // 0x28
207 Dev->Width = LCD32_NATIVE_H;
208 Dev->Height = LCD32_NATIVE_W;
209 Dev->Orientation = LCD32_ORIENTATION_LANDSCAPE; // Correct the state
210 break;
211 }
212 LCD32Log("[LCD32Init] Orientation: %d, W: %d, H: %d, MADCTL: 0x%02X", Dev->Orientation, Dev->Width, Dev->Height, madctl_val);
213
214 // ILI9341 INITIALIZATION SEQUENCE
215 LCD32StartTransaction(Dev);
216
217 // Power Control A
218 LCD32WriteCmd(Dev, ILI9341_POWER_CONTROL_A);
219 LCD32WriteData(Dev, 0x39);
220 LCD32WriteData(Dev, 0x2C);
221 LCD32WriteData(Dev, 0x00);
222 LCD32WriteData(Dev, 0x34);
223 LCD32WriteData(Dev, 0x02);

228 LCD32WriteData(Dev, 0xC1);
229 LCD32WriteData(Dev, 0x30);

233 LCD32WriteData(Dev, 0x85);
234 LCD32WriteData(Dev, 0x00);
235 LCD32WriteData(Dev, 0x78);
236
237 // Driver Timing Control B
238 LCD32WriteCmd(Dev, ILI9341_DRIVER_TIMING_CTRL_B);
239 LCD32WriteData(Dev, 0x00);
240 LCD32WriteData(Dev, 0x00);

242 // Power On Sequence Control
243 LCD32WriteCmd(Dev, ILI9341_POWER_ON_SEQ_CTRL);
244 LCD32WriteData(Dev, 0x64);

413void LCD32FlushCanvasFast(LCD32Dev_t * Dev){
414 // LCD32Entry("LCD32FlushCanvasFast(%p)", Dev);
415
416 if(IsNull(Dev) || IsNull(Dev->Canvas)){
417 return;
418 }
419
420 #if (LCD32_THREAD_SAFE_EN == 1)
421 if (xSemaphoreTake(Dev->mutex, portMAX_DELAY) != pdTRUE) {
422 LCD32Err("[LCD32FlushCanvasFast] Failed to take mutex");
423 return;
424 }
425 #endif
426
428 LCD32SetAddressWindow(Dev, 0, 0, Dev->Width, Dev->Height);
429
431 LCD32StartTransaction(Dev);
432 LCD32SetDataTransaction(Dev);
433
435 {
436 P16Dev_t* P16Dev = &(Dev->P16Com);
437 P16Data_t* DataArr = (P16Data_t*)Dev->Canvas;
438 P16Size_t Size = Dev->Width * Dev->Height;
439
440 #if (P16COM_INIT_CHECK_EN == 1)
441 if( !((P16Dev->StatusFlag) & P16COM_INITIALIZED) ){
442 LCD32Err("[LCD32FlushCanvasFast] P16Com not initialized!");
443 goto cleanup;
444 }
445 #endif
446
447 if(IsNull(DataArr) || IsNotPos(Size)){
448 LCD32Err("[LCD32FlushCanvasFast] Canvas is NULL or Size not valid!");
449 goto cleanup;
450 }
451
452 #if (P16COM_DB_NORMAL_OUTPUT_EN == 0)
453 IOConfigAsOutput(P16Dev->DatIOMask, -1, -1);
454 #endif
455
456 REPN(j, Size){
457 P16Data_t currentData = DataArr[j];
458 uint8_t low_byte = currentData & 0xFF;
459 uint8_t high_byte = (currentData >> 8) & 0xFF;
460
461 uint64_t MaskSet = P16Dev->Lut->LutLow[low_byte].setMask | P16Dev->Lut->LutHigh[high_byte].setMask;
462 uint64_t MaskClr = P16Dev->Lut->LutLow[low_byte].clrMask | P16Dev->Lut->LutHigh[high_byte].clrMask;
463
464 IOSet(MaskSet);
465 IOClr(MaskClr);
466 P16MakeWritePulse(P16Dev);
467 }
468
469 #if (P16COM_DB_NORMAL_OUTPUT_EN == 0)
470 IOConfigAsInput(P16Dev->DatIOMask, -1, -1);
471 #endif
472 }
473
474cleanup:
475 LCD32StopTransaction(Dev);
476
477 #if (LCD32_THREAD_SAFE_EN == 1)
478 xSemaphoreGive(Dev->mutex);
479 #endif
480}

521DefaultRet_t LCD32DrawLine(LCD32Dev_t *Dev, Dim_t r0, Dim_t c0, Dim_t r1, Dim_t c1, Color_t Color) {
522 if (IsNull(Dev)) return STAT_ERR_NULL;
523
524 int16_t dx = abs(c1 - c0), sx = c0 < c1 ? 1 : -1;
525 int16_t dy = -abs(r1 - r0), sy = r0 < r1 ? 1 : -1;
526 int16_t err = dx + dy, e2;
527
528 while (1) {
529 LCD32SetCanvasPixel(Dev, r0, c0, Color);
530 if (r0 == r1 && c0 == c1) break;
531 e2 = 2 * err;
532 if (e2 >= dy) { err += dy; c0 += sx; }
533 if (e2 <= dx) { err += dx; r0 += sy; }
534 }
535 return STAT_OKE;
536}

546DefaultRet_t LCD32DrawThickLine(LCD32Dev_t *Dev, Dim_t r0, Dim_t c0, Dim_t r1, Dim_t c1, Color_t Color, Dim_t Thickness) {
547 if (IsNull(Dev)) return STAT_ERR_NULL;
548 if (Thickness <= 1) return LCD32DrawLine(Dev, r0, c0, r1, c1, Color);
549
550 // Advanced implementation using polygon rendering for sharp corners and flat ends
551 int16_t dx = c1 - c0;
552 int16_t dy = r1 - r0;
553 float len = sqrtf(dx * dx + dy * dy);
554 if (len == 0) return STAT_OKE;
555
556 // Calculate the perpendicular vector
557 float nx = -dy / len;
558 float ny = dx / len;
559
560 float half_thick = (float)Thickness / 2.0f;
561
562 // Calculate the 4 corners of the rectangle representing the thick line
563 LCDPoint_t p[4];
564 p[0] = (LCDPoint_t){ .row = (Dim_t)roundf(r0 - ny * half_thick), .col = (Dim_t)roundf(c0 - nx * half_thick) };
565 p[1] = (LCDPoint_t){ .row = (Dim_t)roundf(r1 - ny * half_thick), .col = (Dim_t)roundf(c1 - nx * half_thick) };
566 p[2] = (LCDPoint_t){ .row = (Dim_t)roundf(r1 + ny * half_thick), .col = (Dim_t)roundf(c1 + nx * half_thick) };
567 p[3] = (LCDPoint_t){ .row = (Dim_t)roundf(r0 + ny * half_thick), .col = (Dim_t)roundf(c0 + nx * half_thick) };
568
569 // Draw the rectangle (quad)
570 LCD32DrawQuad(Dev, p, Color);
571
572 return STAT_OKE;
573}

618 if (IsNull(Dev) || IsNull(Points) || N < 3) return STAT_ERR_INVALID_ARG;
619
620 Dim_t minRow = Points[0].row, maxRow = Points[0].row;
621 for (size_t i = 1; i < N; i++) {
622 if (Points[i].row < minRow) minRow = Points[i].row;
623 if (Points[i].row > maxRow) maxRow = Points[i].row;
624 }
625 if (minRow < 0) minRow = 0;
626 if (maxRow >= Dev->Height) maxRow = Dev->Height - 1;
627
628 int32_t interX[128]; // Max 128 intersections
629
630 for (Dim_t y = minRow; y <= maxRow; y++) {
631 size_t interCount = 0;
632 for (size_t i = 0; i < N; i++) {
633 size_t j = (i + 1) % N;
634 int32_t x0 = Points[i].col, y0 = Points[i].row;
635 int32_t x1 = Points[j].col, y1 = Points[j].row;
636
637 if (y0 == y1) continue; // Horizontal edge
638 if (y0 > y1) { // Swap
639 int32_t tx = x0; x0 = x1; x1 = tx;
640 int32_t ty = y0; y0 = y1; y1 = ty;
641 }
642
643 if (y >= y0 && y < y1) {
644 // Calculate intersection
645 int32_t x = x0 + (int32_t)((int64_t)(x1 - x0) * (y - y0) / (y1 - y0));
646 if (interCount < 128) interX[interCount++] = x;
647 }
648 }
649
650 // Sort intersections using qsort for better performance
651 qsort(interX, interCount, sizeof(int32_t), compare_int32);
652
653 // Fill pairs
654 for (size_t k = 0; k + 1 < interCount; k += 2) {
655 Dim_t xStart = (Dim_t)interX[k];
656 Dim_t xEnd = (Dim_t)interX[k+1];
657 if (xStart < 0) xStart = 0;
658 if (xEnd >= Dev->Width) xEnd = Dev->Width - 1;
659
660 for (Dim_t x = xStart; x <= xEnd; x++) {
661 LCD32SetCanvasPixel(Dev, y, x, Color);
662 }
663 }
664 }
665 return STAT_OKE;
666}

668DefaultRet_t LCD32DrawChar(LCD32Dev_t *Dev, Dim_t r, Dim_t c, char Ch, const GFXfont *Font, Color_t Color) {
669 if (IsNull(Dev) || IsNull(Font)) return STAT_ERR_NULL;
670 if (Ch < Font->first || Ch > Font->last) return STAT_ERR_INVALID_ARG;
671
672 GFXglyph *glyph = &Font->glyph[Ch - Font->first];
673 uint8_t *bitmap = Font->bitmap;
674
675 uint16_t bo = glyph->bitmapOffset;
676 uint8_t w = glyph->width, h = glyph->height;
677 int8_t xo = glyph->xOffset, yo = glyph->yOffset;
678
679 uint8_t bits = 0, bitCount = 0;
680
681 for (Dim_t yy = 0; yy < h; yy++) {
682 for (Dim_t xx = 0; xx < w; xx++) {
683 if (bitCount == 0) {
684 bits = bitmap[bo++];
685 bitCount = 8;
686 }
687 if (bits & 0x80) {
688 LCD32SetCanvasPixel(Dev, r + yo + yy, c + xo + xx, Color);
689 }
690 bits <<= 1;
691 bitCount--;
692 }
693 }
694 return STAT_OKE;
695}

697DefaultRet_t LCD32DrawText(LCD32Dev_t *Dev, Dim_t r, Dim_t c, const char *Str, const GFXfont *Font, Color_t Color) {
698 if (IsNull(Dev) || IsNull(Str) || IsNull(Font)) return STAT_ERR_NULL;
699
700 Dim_t cursor_r = r, cursor_c = c;
701 while (*Str) {
702 char ch = *Str++;
703 if (ch == '\n') {
704 cursor_c = c;
705 cursor_r += Font->yAdvance;
706 continue;
707 }
708 if (ch < Font->first || ch > Font->last) continue;
709
710 GFXglyph *glyph = &Font->glyph[ch - Font->first];
711
712 // Wrap text
713 if (cursor_c + glyph->xAdvance >= Dev->Width) {
714 cursor_c = c;
715 cursor_r += Font->yAdvance;
716 }
717
718 LCD32DrawChar(Dev, cursor_r, cursor_c, ch, Font, Color);
719 cursor_c += glyph->xAdvance;
720 }
721 return STAT_OKE;
722}
