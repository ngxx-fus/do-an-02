102void lcdDrawCaro(uint8_t type) {
103 // 1. Fill the canvas background with white
104 lcd32FillCanvas(lcd, 0xFFFF);
105
106 // Define the color for the grid lines (black)
107 color_t gridColor = COLOR_GIRD;
108
109 // 2. Get grid info from lookup table
110 gridInfo_t grid = gridInfoTable[type];
111
112 // 3. Draw horizontal lines
113 for (dim_t i = 1; i < grid.rowNum; i++) {
114 dim_t y = i * grid.rowStep;
115 lcd32DrawLine(lcd, y, 0, y, LCD32_MAX_COL - 1, gridColor);
116 }
117
118 // 4. Draw vertical lines
119 for (dim_t j = 1; j < grid.colNum; j++) {
120 dim_t x = j * grid.colStep;
121 lcd32DrawLine(lcd, 0, x, LCD32_MAX_ROW - 1, x, gridColor);

266/// @brief Task to handle screen updates, rendering, and power states based on event flags.
267/// @param pv Task parameter (unused).
268void screenControlTask(void * pv){
269 WAIT_SYSTEM_INIT_COMPLETED();
270
271 __entry("screenControlTask()");
272
273 flag_t localScreenFlag;
274 while(!IS_SYSTEM_STOPPED){
275 PERFORM_IN_CRITICAL(&screenFlagMutex,
276 localScreenFlag = screenFlag;
277 screenFlag = 0;
278 );
279
280 while (__isnot_zero(localScreenFlag)){
281 if(__hasFlagBitSet(localScreenFlag, SCREEN_RENDER)){
282 __clearFlagBit(localScreenFlag, SCREEN_RENDER);
283 lcd32FlushCanvas(lcd);
284 }
285 if(__hasFlagBitSet(localScreenFlag, SCREEN_TURN_ON)){
286 __clearFlagBit(localScreenFlag, SCREEN_TURN_ON);
287 lcd32WakeFromSleep(lcd);
288 }
289 if(__hasFlagBitSet(localScreenFlag, SCREEN_TURN_OFF)){
290 __clearFlagBit(localScreenFlag, SCREEN_TURN_OFF);
291 lcd32PutToSleep(lcd);
292 }
293 if(__isnot_zero(localScreenFlag)){
294 __sys_err("[screenControlTask] Warning: Unrecognized screenFlag bits: 0x%08x", localScreenFlag);
295 localScreenFlag = 0;
296 }
297 }
298
299 vTaskDelay(0);
300 }
301
302 __exit("screenControlTask()");
303}
