104static inline __attribute__((always_inline))
105void __lcd32GPIOSetState(uint32_t pin, uint32_t state) {
106 if (pin < 32) {
107 if (state)
108 GPIO.out_w1ts = (1UL << pin); // Set bit
109 else
110 GPIO.out_w1tc = (1UL << pin); // Clear bit
111 } else {
112 uint32_t shift = pin - 32;
113 if (state)
114 GPIO.out1_w1ts.val = (1UL << shift);
115 else
116 GPIO.out1_w1tc.val = (1UL << shift);
117 }
118}

124def __lcd32GPIOGetState(uint32_t pin) {
125 if (pin < 32) {
126 return (GPIO.in >> pin) & 0x1;
127 } else {
128 return (GPIO.in1.val >> (pin - 32)) & 0x1;
129 }
