109/// @brief [Internal] Set GPIO output state (supports GPIO0–GPIO47).
110static inline __attribute__((always_inline))
111void __lcd32GPIOSetState(uint32_t pin, uint32_t state) {
112 if (pin < 32) {
113 if (state)
114 GPIO.out_w1ts = (1UL << pin); // Set bit
115 else
116 GPIO.out_w1tc = (1UL << pin); // Clear bit
117 } else {
118 uint32_t shift = pin - 32;
119 if (state)
120 GPIO.out1_w1ts.val = (1UL << shift);
121 else
122 GPIO.out1_w1tc.val = (1UL << shift);
123 }
124}

126/// @brief [Internal] Get GPIO state via GPIO struct with range check.
127/// @param pin GPIO number (0–47)
128/// @return 1 if HIGH, 0 if LOW.
129static inline __attribute__((always_inline))
130def __lcd32GPIOGetState(uint32_t pin) {
131 if (pin < 32) {
132 return (GPIO.in >> pin) & 0x1;
133 } else {
134 return (GPIO.in1.val >> (pin - 32)) & 0x1;
135 }
136}
