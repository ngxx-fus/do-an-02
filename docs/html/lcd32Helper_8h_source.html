386/// @brief [Internal] Set GPIO output state (supports GPIO0GPIO47).
387static inline __attribute__((always_inline))
388void __lcd32GPIOSetState(uint32_t pin, uint32_t state) {
389 if (pin < 32) {
390 if (state)
391 GPIO.out_w1ts = (1UL << pin); // Set bit
392 else
393 GPIO.out_w1tc = (1UL << pin); // Clear bit
394 } else {
395 uint32_t shift = pin - 32;
396 if (state)
397 GPIO.out1_w1ts.val = (1UL << shift);
398 else
399 GPIO.out1_w1tc.val = (1UL << shift);
400 }
401}

408 if (pin < 32) {
