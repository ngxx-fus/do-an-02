69unsigned int genRandNum(unsigned int seed_input) {
70 // --- 1. Persistent state ---
71 static uint32_t last_known_random_state = 104729u; // initialized with a prime
72 static int last_known_seed_input = 0;
73
74 uint32_t x;
75
76 // --- 2. Select initial seed ---
77 if (last_known_seed_input == seed_input) {
78 x = last_known_random_state;
79 } else {
80 // start from new seed, add prime noise
81 x = (uint32_t)seed_input ^ 0xA5A5A5A5u;
82 }
83 last_known_seed_input = seed_input;
84
85 // --- 3. Prime constants for nonlinear mixing ---
86 const uint32_t P1 = 7919u;
87 const uint32_t P2 = 104729u;
88 const uint32_t P3 = 65537u;
89 const uint32_t P4 = 439u;
90 const uint32_t P5 = 1223u;
91
92 // --- 4. Chaotic transformations ---
93 x = (x * P1 + P3) ^ (x >> 11);
94 x = (x << 7) - (x >> 5) + P4;
95 x ^= (x * P2);
96 x = (x ^ (x >> 17)) + (x << 13);
97
98 // --- 5. Differential and derivative noise ---
99 uint32_t dx = x - ((x >> 1) | (x << 31));
100 uint32_t ddx = dx ^ (dx >> 3) ^ (dx << 5);
101
102 // --- 6. Amplify chaos using primes ---
103 x ^= (ddx * P5);
104 x ^= (dx << 9) | (dx >> 23);
105
106 // --- 7. Update persistent state ---
107 last_known_random_state = x ^ (x >> 16) ^ (x << 7) ^ P2;
108
109 // --- 8. Return full 32-bit random number ---
110 return last_known_random_state;
111}
