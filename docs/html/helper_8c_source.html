6unsigned int genRandNum(unsigned int seed_input) {
7 // --- 1. Persistent state ---
8 static uint32_t last_known_random_state = 104729u; // initialized with a prime
9 static int last_known_seed_input = 0;
10
11 uint32_t x;
12
13 // --- 2. Select initial seed ---
14 if (last_known_seed_input == seed_input) {
15 x = last_known_random_state;
16 } else {
17 // start from new seed, add prime noise
18 x = (uint32_t)seed_input ^ 0xA5A5A5A5u;
19 }
20 last_known_seed_input = seed_input;
21
22 // --- 3. Prime constants for nonlinear mixing ---
23 const uint32_t P1 = 7919u;
24 const uint32_t P2 = 104729u;
25 const uint32_t P3 = 65537u;
26 const uint32_t P4 = 439u;
27 const uint32_t P5 = 1223u;
28
29 // --- 4. Chaotic transformations ---
30 x = (x * P1 + P3) ^ (x >> 11);
31 x = (x << 7) - (x >> 5) + P4;
32 x ^= (x * P2);
33 x = (x ^ (x >> 17)) + (x << 13);
34
35 // --- 5. Differential and derivative noise ---
36 uint32_t dx = x - ((x >> 1) | (x << 31));
37 uint32_t ddx = dx ^ (dx >> 3) ^ (dx << 5);
38
39 // --- 6. Amplify chaos using primes ---
40 x ^= (ddx * P5);
41 x ^= (dx << 9) | (dx >> 23);
42
43 // --- 7. Update persistent state ---
44 last_known_random_state = x ^ (x >> 16) ^ (x << 7) ^ P2;
45
46 // --- 8. Return full 32-bit random number ---
47 return last_known_random_state;
48}
