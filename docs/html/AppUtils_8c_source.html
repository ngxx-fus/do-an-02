23 static uint32_t last_known_random_state = 104729u; // initialized with a prime
24 static int last_known_seed_input = 0;
25

27
28 // --- 2. Select initial seed ---
29 if (last_known_seed_input == SeedInput) {
30 x = last_known_random_state;
31 } else {
32 // start from new seed, add prime noise
33 x = (uint32_t)SeedInput ^ 0xA5A5A5A5u;
34 }
35 last_known_seed_input = SeedInput;
36
37 // --- 3. Prime constants for nonlinear mixing ---
38 const uint32_t P1 = 7919u;
39 const uint32_t P2 = 104729u;
40 const uint32_t P3 = 65537u;
41 const uint32_t P4 = 439u;
42 const uint32_t P5 = 1223u;
43
44 // --- 4. Chaotic transformations ---
45 x = (x * P1 + P3) ^ (x >> 11);
46 x = (x << 7) - (x >> 5) + P4;
47 x ^= (x * P2);
48 x = (x ^ (x >> 17)) + (x << 13);
49
50 // --- 5. Differential and derivative noise ---
51 uint32_t dx = x - ((x >> 1) | (x << 31));
52 uint32_t ddx = dx ^ (dx >> 3) ^ (dx << 5);
53
54 // --- 6. Amplify chaos using primes ---
55 x ^= (ddx * P5);
56 x ^= (dx << 9) | (dx >> 23);
57
58 // --- 7. Update persistent state ---
59 last_known_random_state = x ^ (x >> 16) ^ (x << 7) ^ P2;
60
61 // --- 8. Return full 32-bit random number ---
