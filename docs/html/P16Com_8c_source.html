12static void P16ComBuildLut(const Pin_t* DatPinArr, P16Lut_t* Lut) {
13 P16Entry("P16ComBuildLut(%p, %p)", DatPinArr, Lut);
14 // Build LUT for low byte (bits 0-7)
15 for (uint32_t i = 0; i < 256; i++) {
16 uint64_t setMask = 0;
17 uint64_t clrMask = 0;
18 for (uint8_t bit = 0; bit < 8; bit++) {
19 if ((i >> bit) & 1) {
20 setMask |= (1ULL << DatPinArr[bit]);
21 } else {
22 clrMask |= (1ULL << DatPinArr[bit]);
23 }
24 }
25 Lut->LutLow[i].setMask = setMask;
26 Lut->LutLow[i].clrMask = clrMask;
27 }
28 // Build LUT for high byte (bits 8-15)
29 for (uint32_t i = 0; i < 256; i++) {
30 uint64_t setMask = 0;
31 uint64_t clrMask = 0;
32 for (uint8_t bit = 0; bit < 8; bit++) {
33 if ((i >> bit) & 1) {
34 setMask |= (1ULL << DatPinArr[bit + 8]);
35 } else {
36 clrMask |= (1ULL << DatPinArr[bit + 8]);
37 }
38 }
39 Lut->LutHigh[i].setMask = setMask;
40 Lut->LutHigh[i].clrMask = clrMask;
41 }
42 P16Log("[P16ComBuildLut] LUTs generated for fast writes.");
43}
