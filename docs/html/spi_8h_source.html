158/// @brief Set CLK pin level based on CPOL and desired logical state (IDLE/ACTIVE)
159/// @param dev Pointer to SPI device descriptor
160/// @param state Desired logical state (SPICLK_IDLE or SPICLK_ACTIVE)
161static inline __attribute__((always_inline))
162void spiSetCLKState(spiDev_t *dev, enum SPI_CLK_STATE state) {
163 uint8_t cpol = __hasFlagBitSet(dev->conf, SPI_CPOL);
164 uint32_t mask = __mask32(dev->clk);
165 if (state == SPICLK_IDLE)
166 (cpol ? (GPIO.out_w1ts = mask) : (GPIO.out_w1tc = mask));
167 else
168 (cpol ? (GPIO.out_w1tc = mask) : (GPIO.out_w1ts = mask));
169}
170
171/// @brief Set GPIO pin to specific level
172/// @param gpio GPIO pin number
173/// @param level Desired level (HIGH/LOW)
174static inline __attribute__((always_inline))
175void spiSetGPIO(uint32_t gpio, uint8_t level) {
176 if (level == LOW)
177 GPIO.out_w1tc = __mask32(gpio);
178 else
179 GPIO.out_w1ts = __mask32(gpio);
180}
