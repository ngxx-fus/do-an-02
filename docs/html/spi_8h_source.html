156static inline __attribute__((always_inline))
157void spiSetCLKState(spiDev_t *dev, enum SPI_CLK_STATE state) {
158 uint8_t cpol = __hasFlagBitSet(dev->conf, SPI_CPOL);
159 uint32_t mask = __mask32(dev->clk);
160 if (state == SPICLK_IDLE)
161 (cpol ? (GPIO.out_w1ts = mask) : (GPIO.out_w1tc = mask));
162 else
163 (cpol ? (GPIO.out_w1tc = mask) : (GPIO.out_w1ts = mask));
164}
